    <p className="mb-3 text-base-content/70">Add a selection column with checkboxes:</p>
```tsx
const { rowSelection } = useVirtualTable({
  data,
  columns,
  enableRowSelection: true,
  getRowId: (row) => row.id,
});

// Selection column:
{
  id: 'select',
  header: ({ table }) => (
    <Checkbox checked={table.getIsAllRowsSelected()} />
  ),
  cell: ({ row }) => (
    <Checkbox checked={row.getIsSelected()} />
  ),
}
```
